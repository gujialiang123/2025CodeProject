原题：CF122A

难度：普及-，1000

算法：暴力 数学

读入后判断一遍即可：

1.能被幸运数字整除。

2.仅包含幸运数字。
```
#include<cstdio>
using namespace std;

int main()
{
    int n;
    scanf("%d",&n);
    if(n%4==0||n%7==0||n%74==0||n%47==0||n%44==0||n%77==0)
    //能被幸运数字整除（77以上的数字不需要判断了）
        return printf("YES"),0;
    while(n!=0)
    {
        register int t=n%10;
        if(t!=4&&t!=7)  return printf("NO"),0;
        n/=10;
    }
    return printf("YES"),0;
}
```
